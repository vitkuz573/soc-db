    #### f) Add tracing middleware for Prometheus metrics

    Add a new middleware that traces Prometheus metrics (request counting, in-flight tracking, duration recording). This middleware runs AFTER the rate limiter but BEFORE the request ID middleware (so it captures rate-limited requests too):

    ```python
    @app.middleware("http")
    async def trace_metrics_middleware(request: Request, call_next):
        """Record Prometheus metrics for every HTTP request.

        Tracks:
        - ``requests_in_flight`` — incremented on entry, decremented on exit
        - ``requests_total`` — incremented per response (labels: method, sanitised path, status)
        - ``request_duration_seconds`` — observed per response
        """
        requests_in_flight.inc()
        start = time.monotonic()
        try:
            response = await call_next(request)
        except BaseException:
            # Still record metrics even on unhandled exceptions
            elapsed = time.monotonic() - start
            path = _sanitise_path(request.url.path)
            requests_total.labels(method=request.method, path=path, status="500").inc()
            request_duration_seconds.labels(method=request.method, path=path, status="500").observe(elapsed)
            requests_in_flight.dec()
            raise
        elapsed = time.monotonic() - start
        path = _sanitise_path(request.url.path)
        status = str(response.status_code)
        requests_total.labels(method=request.method, path=path, status=status).inc()
        request_duration_seconds.labels(method=request.method, path=path, status=status).observe(elapsed)
        requests_in_flight.dec()
        return response
    ```

    The `try/except BaseException` block ensures we decrement the in-flight gauge even if an unhandled exception propagates (otherwise the gauge would leak — it would stay incremented forever).

    ```python
        """Sanitise a URL path for use as a Prometheus label value.

        Replaces dynamic path segments (variables after /v1/chips/ or /v1/export/)
        with a static placeholder ``{param}`` to bound label cardinality.

        Args:

        if path.startswith("/v1/chips/") and path.count("/") == 3:

        # Export: /v1/export/{fmt}
        if path.startswith("/v1/export/") and path.count("/") == 3:

        return path
    ```

    ### 2. tests/benchmark/test_otel_overhead.py — OTel overhead benchmark

    Create a new benchmark test file that measures latency with and without OpenTelemetry to verify overhead is under 5%.

    The benchmark must:

    ```python
    """Benchmark measuring OpenTelemetry overhead on API request latency.

    Runs a simple ASGI request with and without OTel instrumentation,
    comparing p50, p95, and p99 latencies.  Overhead must be under 5%
    at p50 (target: < 3%).
    """

    from __future__ import annotations

    import statistics
    import time

    import pytest
    from httpx import ASGITransport, AsyncClient

    from api.main import app


    @pytest.fixture
    def client():
        transport = ASGITransport(app=app)
        return AsyncClient(transport=transport, base_url="http://test")


    @pytest.mark.asyncio
    @pytest.mark.benchmark
    async def test_otel_overhead(client):
        """Measure OTel overhead by comparing before/after request latencies.

        Warmup: 10 requests
        Measure without OTel: 200 requests to /health (excluded from tracing)
        Measure with OTel: 200 requests to /v1/chips (full tracing)

        The /health path is excluded from OpenTelemetry instrumentation,
        so it serves as the baseline.  The /v1/chips path goes through
        the full tracing pipeline.
        """
        # Warmup
        for _ in range(10):
            await client.get("/health")

        # Baseline: /health (no tracing)
        health_times: list[float] = []
        for _ in range(200):
            start = time.monotonic()
            resp = await client.get("/health")
            elapsed = time.monotonic() - start
            health_times.append(elapsed)
            assert resp.status_code in (200, 503)  # 503 = not ready in JSON mode

        # Traced: /v1/chips (full OTel tracing)
        chip_times: list[float] = []
        for _ in range(200):
            start = time.monotonic()
            resp = await client.get("/v1/chips?limit=10")
            elapsed = time.monotonic() - start
            chip_times.append(elapsed)
            assert resp.status_code == 200

        # Compute statistics
        def pctile(data: list[float], p: float) -> float:
            idx = int(len(data) * p / 100)
            return sorted(data)[min(idx, len(data) - 1)]

        h_p50 = pctile(health_times, 50) * 1000  # ms
        h_p95 = pctile(health_times, 95) * 1000
        h_p99 = pctile(health_times, 99) * 1000
        c_p50 = pctile(chip_times, 50) * 1000
        c_p95 = pctile(chip_times, 95) * 1000
        c_p99 = pctile(chip_times, 99) * 1000

        overhead_pct = ((c_p50 - h_p50) / h_p50) * 100 if h_p50 > 0 else 0.0

        # Print results for CI visibility
        print(f"\n{'='*60}")
        print(f"  OTel Overhead Benchmark")
        print(f"{'='*60}")
    ```
