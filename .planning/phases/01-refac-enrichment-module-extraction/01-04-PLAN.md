    ```python
    # -- Functions that remain in common.py (not enrichment) --
    def fetch(...): ...
    def extract_int(...): ...
    def extract_freq(...): ...
    def extract_process(...): ...
    def clean(...): ...  # re-exported from enrich

    def write_vendor_file(...): ...
    def extract_model(...): ...

    # -- Enrichment controller (orchestrates domain modules) --
    def enrich_one(chip: dict[str, Any]) -> dict[str, Any]:
        \"\"\"Enrich a single chip record by applying per-domain inference modules.\"\"\"
        # Cleanup
        ann = re.compile(r"\s*\(now\s+[^)]*?\)")
        for k in ("name", "model"):
            if chip.get(k):
                cleaned = ann.sub("", chip[k]).strip()
                if cleaned != chip[k]:
                    chip[k] = cleaned
        if not chip.get("model"):
            name = chip.get("name", "")
            if re.match(r"^[A-Za-z0-9][A-Za-z0-9/\-.\s]{1,30}$", name) and re.search(r"\d", name):
                chip["model"] = name
            else:
                chip["model"] = chip.get("id", "unknown")
        
        # Pipeline — per-domain inferences
        vk = VENDOR_KNOWLEDGE.get(chip.get("vendor", ""), {})
        model_upper = chip.get("model", "").upper()
        
        from soc_db.enrich.cpu import infer_cpu
        from soc_db.enrich.memory import infer_memory
        from soc_db.enrich.process import infer_process
        from soc_db.enrich.gpu import infer_gpu
        from soc_db.enrich.year import infer_year
        from soc_db.enrich.modem import infer_modem
        from soc_db.enrich.npu import infer_npu
        from soc_db.enrich.connectivity import infer_wifi, infer_bluetooth
        from soc_db.enrich.storage import infer_storage
        from soc_db.enrich.aliases import infer_aliases
        from soc_db.enrich.scoring import compute_completeness
        
        infer_cpu(chip)
        infer_memory(chip)         # first pass: clock/bus from type
        infer_process(chip)        # first pass: model-based lookup
        infer_gpu(chip)            # first pass: model-based lookup
        
        # Year validation + inference
        y_chk = chip.get("year")
        if y_chk and (y_chk < 2003 or y_chk > 2026):
            chip["year"] = None
        year = chip.get("year")
        if not year:
            inferred = infer_year(chip)
            if inferred:
                chip["year"] = inferred
        
        infer_process(chip)        # second pass: year-based fallback
        infer_memory(chip)         # second pass: type/clock/bus from year
        infer_storage(chip)
        infer_gpu(chip)            # second pass: vendor defaults from year
        infer_npu(chip)
        infer_modem(chip)
        infer_wifi(chip)
        infer_bluetooth(chip)
        infer_aliases(chip)
        compute_completeness(chip)
        
        if not chip.get("sources"):
            chip["sources"] = {}
        chip["updated"] = "2026-06-21"
        return chip
    
    def enrich_all(chips: list[dict[str, Any]]) -> list[dict[str, Any]]:
        \"\"\"Apply enrich_one to every chip in a list.\"\"\"
        return [enrich_one(c) for c in chips]
    ```
